00064                 static Eigen::Matrix<double,2,qOrder> abcsissasAndWeights(){
00065                         
00066                         // Create and fill the Jacobi upper triangular matrix  
00067                         Eigen::Matrix<double,qOrder,qOrder> jUT = Eigen::Matrix<double,qOrder,qOrder>::Zero();
00068                         for (int n=0;n<qOrder;++n){
00069                                 int k = n+1;
00070                                 jUT(n,n) = -Derived::b(k)/Derived::a(k);
00071                                 if (n<qOrder-1) {
00072                                         jUT(n,n+1)= std::sqrt(Derived::c(k+1)/Derived::a(k)/Derived::a(k+1));
00073                                 }
00074                         }
00075                         
00076                         // Create a SelfAdjointEigenSolver from the self-adjoint part of jUT
00077                         Eigen::SelfAdjointEigenSolver<Eigen::Matrix<double,qOrder,qOrder> > es(jUT.template selfadjointView<Eigen::Upper>());
00078                         Eigen::Matrix<double,qOrder,1>          eigenVal = es.eigenvalues();
00079                         Eigen::Matrix<double,qOrder,qOrder>     eigenVec = es.eigenvectors();
00080                         
00081                         // Insert each abscissa-weight pair in std::set for automatic sorting with increasing abscissas
00082                         typedef std::set<std::pair<double,double> > EigenSetType;
00083                         typedef EigenSetType::const_iterator            EigenSetIterType;
00084                         EigenSetType eigenSet;
00085                         for (int k=0;k<qOrder;++k){
00086                                 eigenSet.insert( std::make_pair( (eigenVal(k)+1.0)*0.5, std::pow(eigenVec(0,k),2) ) );
00087                         }
00088                         
00089                         // Copy the sorted values to abscissas_ and weigts_
00090                         enum {dim=1};
00091                         Eigen::Matrix<double,dim+1,qOrder> abcsissasAndWeights_;
00092                         int k=0;
00093                         for (EigenSetIterType iter=eigenSet.begin(); iter!=eigenSet.end();++iter){
00094                                 abcsissasAndWeights_(0,k)= iter->first;
00095                                 abcsissasAndWeights_(1,k)= iter->second;
00096                                 ++k;
00097                         }
00098                         
00099                         std::cout<< "Quadrature rule of order "<<qOrder<<" (weights in last row):"<<std::endl;
00100                         std::cout<<std::setprecision(15)<<std::scientific<<abcsissasAndWeights_<<std::endl;
00101                         
00102                         return abcsissasAndWeights_;
00103                 }
