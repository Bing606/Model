00040         BinaryFileReader(const std::string& filename) : _bytes_in_Type(sizeof (Type) ){
00041                 
00042                 // open file with the ios::ate flag in order to set the get pointer at the end of the file. 
00043                 std::ifstream file (filename.c_str(), std::ios::in|std::ios::binary|std::ios::ate);
00044                 
00045                 if (file.is_open()){
00046                 
00047                         // Since the get pointer is at the end of the file, call file.tellg() to obtain the number of bytes in the file
00048                         _bytes_in_file = file.tellg();
00049                         std::cout << "File '"<< filename <<"' contains:"<<std::endl;
00050                         std::cout << "  bytes in file : "     << _bytes_in_file <<std::endl;
00051                         std::cout << "  bytes per element:  " << _bytes_in_Type  <<std::endl;
00052                         
00053                         // The number of bytes in the file must be a multiple of the number of bytes in Type 
00054                         assert((_bytes_in_file%_bytes_in_Type)==0 && " Incorrect format.");
00055                         
00056                         // Calculate how many element of type Type are in the file
00057                         _array_size = _bytes_in_file/_bytes_in_Type;
00058                         std::cout << "  array size:  "<< _array_size  <<std::endl;
00059                         
00060                         // resize memblock with _array_size
00061                         memblock = new Type [_array_size];
00062                         
00063                         // bring back the get pointer to the beginning of the file
00064                         file.seekg (0, std::ios::beg);
00065                         
00066                         // read the file into memblock
00067                         file.read (reinterpret_cast<char*>(memblock), _bytes_in_file);
00068                         
00069                         // close the file
00070                         file.close();
00071                         
00072                         success=true;
00073                 }
00074                 else{
00075                         success=false;
00076                         std::cout << "Unable to open file "<<filename<<std::endl;
00077                 }
