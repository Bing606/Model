00059                 gsl_vector * beta = gsl_vector_alloc (n);
00060 //              gsl_vector_complex * beta = gsl_vector_complex_alloc (n);
00061                 gsl_matrix_complex * evec = gsl_matrix_complex_alloc(n,n);
00062                 
00063                 
00064                 gsl_eigen_genv (&A.matrix, &B.matrix, alpha, beta, evec, w);
00065                 
00066                 
00067                 for (int col = 0; col < n; ++col) {
00068                         gsl_complex alpha_i = gsl_vector_complex_get (alpha, col);
00069                         double beta_i       = gsl_vector_get(beta,col);
00070                         D(col) = std::complex<double>(GSL_REAL(alpha_i)/beta_i,GSL_IMAG(alpha_i)/beta_i);
00071                         
00072                         for (int row = 0; row < n; ++row) {
00073                                 gsl_complex Vij = gsl_matrix_complex_get(evec,row,col);
00074                                 V(row,col) = std::complex<double>(GSL_REAL(Vij),GSL_IMAG(Vij));
00075                         }
00076                 }
00077                 
00078                 // 2 - frees the memory associated with the workspace w
00079                 gsl_eigen_genv_free (w);
