00251                 {

00253                                                 
00254                         Kn(0,0)= wv *((phi(ai,0)*c11*phi(bi,0)) + (phi(ai,1)*c44*phi(bi,1)) +

00256                         Kn(0,1)= wv *((phi(ai,0)*c12*phi(bi,1)) + (phi(ai,1)*c44*phi(bi,0)) );
00257                         Kn(0,2)= wv *((phi(ai,0)*c12*phi(bi,2)) + (phi(ai,2)*c44*phi(bi,0)) );
00258                         
00259                         Kn(1,0)= wv *((phi(ai,1)*c12*phi(bi,0)) + (phi(ai,0)*c44*phi(bi,1)) );
00260                         Kn(1,1)= wv *((phi(ai,1)*c11*phi(bi,1)) + (phi(ai,0)*c44*phi(bi,0)) + 
00261                                                    (phi(ai,2)*c44*phi(bi,2)) );
00262                         Kn(1,2)= wv *((phi(ai,1)*c12*phi(bi,2)) + (phi(ai,2)*c44*phi(bi,1)) );
00263                         
00264                         Kn(2,0)= wv *((phi(ai,2)*c12*phi(bi,0)) + (phi(ai,0)*c44*phi(bi,2)) );
00265                         Kn(2,1)= wv *((phi(ai,2)*c12*phi(bi,1)) + (phi(ai,1)*c44*phi(bi,2)) );
00266                         Kn(2,2)= wv *((phi(ai,2)*c11*phi(bi,2)) + (phi(ai,1)*c44*phi(bi,1)) +

00280                         //------- The stress components ---------------
00281                         Eigen::Matrix<double,dim,dim> stress;   
00282                         
00283                         double em = lambda*uprim.trace();

00285                         stress(0,0) = em + (2.0e0*mu*uprim(0,0));
00286                         stress(1,1) = em + (2.0e0*mu*uprim(1,1));
00287                         stress(2,2) = em + (2.0e0*mu*uprim(2,2));
00288                         
00289                         stress(0,1) = mu*(uprim(0,1)+uprim(1,0));               stress(1,0) = stress(0,1);
00290                         stress(0,2) = mu*(uprim(2,0)+uprim(0,2));               stress(2,0) = stress(0,2);
00291                         stress(1,2) = mu*(uprim(1,2)+uprim(2,1));               stress(2,1) = stress(1,2);
00292                         
00293                         
00294                         return stress;
