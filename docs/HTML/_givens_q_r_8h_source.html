00017         template<int N, int ImJ, int NmJ>
00018         struct GivensQRkernel{
00019 
00020                 enum {J=N-NmJ}; // J=N-(N-J)
00021                 enum {I=ImJ+J}; // I=(I-J)+J
00022                 enum {NmI=N-I}; // 
00023                 
00024                 static  void step(Eigen::Matrix<double,N,N>& Q, Eigen::Matrix<double,N,N>& R){
00025                         Eigen::Matrix<double,2,2> G(Givens::Q<double>(R(I,J),R(I+1,J)));
00026                         R.template block<2,NmJ>(I,J)=(G*R.template block<2,NmJ>(I,J)).eval();                   // use .eval() to avoid aliasing
00027                         Q.template block<N,2>(0,I)=(Q.template block<N,2>(0,I)*G.transpose()).eval();   // use .eval() to avoid aliasing
00028                 }
00029         };

00031         template<int N, int ImJ, int NmJ>
00032         struct GivensQRstep{
00033                 /* Define ImJ=I-J 
00034                  *    and NmJ=N-J
00035                  * therefore: J=N-NmJ 
00036                  *            I=ImJ+J=ImJ+N-NmJ
00037                  * when the current element is trasformed call recursivly decreasing ImJ by one, that is
00038                  * GivensQRstep<N,ImJ-1,NmJ>
00039                  */
00040                 static  void step(Eigen::Matrix<double,N,N>& Q, Eigen::Matrix<double,N,N>& R){
00041                         Eigen::Matrix<double,2,2> G(Givens::Q<double>(R(ImJ+N-NmJ,N-NmJ),R(ImJ+N-NmJ+1,N-NmJ)));
00042                         GivensQRkernel<N,ImJ,NmJ>::step(Q,R);
00043                         GivensQRstep<N,ImJ-1,NmJ>::step(Q,R);
00044                 }
00045         };
