00057                 /* read *************************************************/              
00058                 void read(const int& frameN){
00059                         // Read edge file T only if it exists, otherwise try to read 0
00060                         edgeFileIsGood=EdgeContainerType::isGood<true>(frameN);
00061                         if (edgeFileIsGood){
00062                                 EdgeContainerType::read<true>(frameN);
00063                         }
00064                         else{
00065                                 EdgeContainerType::read<true>(0);                                       
00066                         }
00067                         
00068                         // Read node file N only if it exists, otherwise try to read 0
00069                         nodeFileIsGood=NodeContainerType::isGood(frameN);

00071                                 NodeContainerType::read(frameN);
00072                         }
00073                         else{

00075                         }
00076                         
00077                         dispFileIsGood=DispContainerType::isGood(frameN);
00078                         if (dispFileIsGood){
00079                                 DispContainerType::read(frameN);
00080                                 
00081                         }
00082                         else{
00083                                 DispContainerType::read(0);
00084                         }
00085                         
00086                         //                      assert(NodeContainerType==DispContainerType && "NUMBER OF NODES IN DISPLACEMENT FILE");
00087                         
00088                         
00089                         edgeVector.clear();
00090                         edgeVector.reserve(EdgeContainerType::size()); // use reserve to speed-up push_back used later
00091                         for (EdgeContainerType::const_iterator iterE=EdgeContainerType::begin(); iterE!=EdgeContainerType::end();++iterE){
00092                                 VertexReader<'N',5,float>::const_iterator iterN1 = NodeContainerType::find(iterE->first.first);
00093                                 VertexReader<'N',5,float>::const_iterator iterN2 = NodeContainerType::find(iterE->first.second);
00094                                 assert(iterN1!=NodeContainerType::end() && "MESH NODE NOT FOUND IN N FILE");
00095                                 assert(iterN2!=NodeContainerType::end() && "MESH NODE NOT FOUND IN N FILE");
00096                                 bool isBoundaryNode1(iterN1->second.operator()(3));
00097                                 bool isBoundaryNode2(iterN2->second.operator()(3));
00098                                 
00099                                 if ( (isBoundaryNode1 && isBoundaryNode2) /*|| showInteriorBoundaryMesh*/ ) {
00100                                         Eigen::Matrix<float,3,4> temp(Eigen::Matrix<float,3,4>::Zero());
00101                                         temp.col(0)=iterN1->second.segment<3>(0);
00102                                         temp.col(1)=iterN2->second.segment<3>(0);
00103                                         if (dispFileIsGood) {
00104                                                 VertexReader<'D',4,float>::iterator iterD1(DispContainerType::find(iterE->first.first));
00105                                                 VertexReader<'D',4,float>::iterator iterD2(DispContainerType::find(iterE->first.second));
00106                                                 assert(iterD1!=DispContainerType::end() && "MESH NODE NOT FOUND IN D FILE");
00107                                                 assert(iterD2!=DispContainerType::end() && "MESH NODE NOT FOUND IN D FILE");
00108                                                 temp.col(2)=iterD1->second.segment<3>(0);
00109                                                 temp.col(3)=iterD2->second.segment<3>(0);
00110                                         }
00111                                         edgeVector.push_back(temp);
00112                                 }
00113                         }
00114                 }
