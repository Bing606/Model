00054                 template <typename ...NodeArgTypes>
00055                 std::pair<bool,size_t> expand(const size_t& i, const size_t& j, const NodeArgTypes&... Args){
00056                         isConstNetworkEdgeType Lij(EdgeFinder<EdgeType>(networkEdgeMapRef).link(i,j));
00057                         assert(Lij.first && "EXPANDING NON-EXISTING LINK.");
00058                         const size_t newID(VertexInsertion<VertexType>(networkVertexMapRef).insert(ExpandingEdge<EdgeType>(*Lij.second),Args...));
00059                         VertexConnection<VertexType,EdgeType>(networkVertexMapRef,networkEdgeMapRef).connect(i,newID,ExpandingEdge<EdgeType>(*Lij.second));
00060                         VertexConnection<VertexType,EdgeType>(networkVertexMapRef,networkEdgeMapRef).connect(newID,j,ExpandingEdge<EdgeType>(*Lij.second));
00061                         VertexConnection<VertexType,EdgeType>(networkVertexMapRef,networkEdgeMapRef).template disconnect<0>(i,j);
00062                         return std::make_pair(true,newID);
00063                 }
