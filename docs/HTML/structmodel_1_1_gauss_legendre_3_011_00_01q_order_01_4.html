template<short unsigned int qOrder>
struct model::GaussLegendre< 1, qOrder >

The P_n(x) = (2n-1)/n * x * P_(n-1) - (n-1)/n * P_(n-2)

gives: A_n = (2*n-1)/n B_n = 0 C_n = (n-1)/n
