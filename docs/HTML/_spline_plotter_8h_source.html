00076                 MatrixDimNc getCircle(const int& k) const {
00077                         MatrixDimNc circle;
00078                         for (int c=0;c<Nc;++c){
00079                                 circle.col(c)=  Eigen::AngleAxisf(2.0f * M_PI/Nc*c,tubeTangents.col(k)) * tubeTangents.col(k).cross(planeNormal);
00080                         }

00096                         
00097                         
00098                         tubeCircles.reserve(Np); // set the capacity of the vector to speed-up push_back
00099                         
00100                         
00101                         //                      radius=radius_in;
00102                         chord = P0T0P1T1BN.col(2)-P0T0P1T1BN.col(0);
00103                         scalarType g = std::pow(chord.norm(),alpha);
00104                         
00105                         for (int k=0;k<Np;++k){
00106                                 scalarType u1=k*1.0/(Np-1);
00107                                 scalarType u2=u1*u1;
00108                                 scalarType u3=u2*u1;
00109                                 
00110                                 // Compute positions along axis
00111                                 tubeAxis.col(k) =   ( 2.0f*u3-3.0f*u2+1.0f) * P0T0P1T1BN.col(0)
00112                                 /*************/ + g*(      u3-2.0f*u2+u1)   * P0T0P1T1BN.col(1)
00113                                 /*************/ +   (-2.0f*u3+3.0f*u2)      * P0T0P1T1BN.col(2)
00114                                 /*************/ + g*(      u3-u2)           * P0T0P1T1BN.col(3);
00115                                 
00116                                 // Compute tangents along axis                          
00117                                 tubeTangents.col(k)= (     ( 6.0f*u2-6.0f*u1)      * P0T0P1T1BN.col(0)
00118                                                                           /*                  */ + g*( 3.0f*u2-4.0f*u1+1.0f) * P0T0P1T1BN.col(1)
00119                                                                           /*                  */ +   (-6.0f*u2+6.0f*u1)      * P0T0P1T1BN.col(2)
00120                                                                           /*                  */ + g*( 3.0f*u2-2.0f*u1)      * P0T0P1T1BN.col(3) ).normalized();        
00121                                 
00122                                 // Compute unit vectors in radial direction orthogonal to axis
00123                                 tubeCircles.push_back(getCircle(k));
00124                         }
00125                         
00126                         
00127                         
00128                         
00129                         
00130                         colorVector = burgers;
00131                         
00132                         if(colorVector(0)<0.0){
00133                                 colorVector*=-1;
00134                         }
00135                         else if(colorVector(0)==0.0){
00136                                 if(colorVector(1)<0.0){
00137                                         colorVector*=-1;
00138                                 }
00139                                 else if(colorVector(1)==0.0){
00140                                         if(colorVector(2)<0.0){
00141                                                 colorVector*=-1;
00142                                         }                                       
00143                                 }
00144                         }
00145                         
00146                         //                      VectorDim colorVector = burgers + VectorDim::Ones(dim) * burgers.norm();                        
00147                         colorVector = (colorVector + VectorDim::Ones(dim) * colorVector.norm()).eval();
00148                         
00149         //              colorVector << 0.0f,0.6f,0.4f;
00150                         colorVector.normalize();
00151                         
00152                 }

00289                 void read(const int& frameN){

00294                         VertexContainerType::read(frameN);
00295                         EdgeContainerType::read<true>(frameN);
00296                         SingleSplinePlotterVectorType::clear(); // clear the current content of sspVector
00297                         SingleSplinePlotterVectorType::reserve(EdgeContainerType::size()); // reserve to speed-up push_back
00298                         for (EdgeContainerType::const_iterator itEdge=EdgeContainerType::begin(); itEdge !=EdgeContainerType::end(); ++itEdge) {

00303                                 
00304                                 
00305                                 Eigen::Matrix<scalarType,dim,6> P0T0P1T1BN;
00306                                 
00307                                 int sourceTfactor(itEdge->second(2*dim));
00308                                 int   sinkTfactor(itEdge->second(2*dim+1));
00309                                 
00310                                 if(deformedConfig){
00311                                         P0T0P1T1BN.col(0) = itSource->second.segment<dim>(7).transpose().template cast<float>();        // source position
00312                                         P0T0P1T1BN.col(2) =   itSink->second.segment<dim>(7).transpose().template cast<float>();        // sink position
00313                                 }
00314                                 else{
00315                                         P0T0P1T1BN.col(0) = itSource->second.segment<dim>(0*dim).transpose().template cast<float>();    // source position
00316                                         P0T0P1T1BN.col(2) =   itSink->second.segment<dim>(0*dim).transpose().template cast<float>();    // sink position
00317                                 }
00318                                 P0T0P1T1BN.col(1) = sourceTfactor*(itSource->second.segment<dim>(1*dim).transpose().template cast<float>());    // source tangent
00319                                 P0T0P1T1BN.col(3) =  -sinkTfactor*(  itSink->second.segment<dim>(1*dim).transpose().template cast<float>());    // sink tangent
00320                                 P0T0P1T1BN.col(4) = itEdge->second.segment<dim>(0*dim).transpose().template cast<float>();              // Burgers vector
00321                                 P0T0P1T1BN.col(5) = itEdge->second.segment<dim>(1*dim).transpose().template cast<float>();              // plane normal
