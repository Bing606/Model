00059         template <typename T>
00060         struct ObservingSet{
00061                 
00062                 typedef std::set<T*> SetType;   // std::set stores const objects so addresses will be const
00063                 typedef typename SetType::const_iterator SetIteratorType;
00064                 friend class ObservedBySet<T>;  // give ObservedBySet<T> access to staticSet
00065 
00066                 /* size **********************************************/
00067                 size_t  size() const {
00069                         return this->staticSet.size();
00070                 }       
00071                 
00072                 /* observedBegin *************************************/
00073                 SetIteratorType observedBegin() const {
00074                         return this->staticSet.begin();
00075                 }
00076                 
00077                 /* observedEnd ***************************************/
00078                 SetIteratorType observedEnd() const {
00079                         return this->staticSet.end();
00080                 }
00081                 
00082                 /* observed ******************************************/
00083                 T& observed(const unsigned int& k){
00084                         assert(k<staticSet.size() && "REQUESTED ELEMENT OUT OF BOUND.");
00085                         SetIteratorType it(staticSet.begin());
00086                         std::advance (it,k);
00087                         return **it;
00088                 }
00089                 
00090         private:
00091                 static SetType staticSet; // the static map containing pointers to T objects
00092 
00093         };
