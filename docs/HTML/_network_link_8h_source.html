00072                         else{ // source and sink are in different subnetworks

00082                                 size_t sourceSNsize(source->pSN()->nodeOrder());
00083                                 size_t   sinkSNsize(  sink->pSN()->nodeOrder());
00084                                 if (sourceSNsize>=sinkSNsize){
00085                                         psn=source->pSN();                                         // redirect psn to the source psn
00086                                         psn->add(this->p_derived());               // add this to the source subnetwork
00087                                         sink->depthFirstExecute(Nmfp,Lmfp,this->psn);   // Transmits 'formSubNetwork' on the sink side
00088                                 }
00089                                 else{
00090                                         psn=sink->pSN();                                       // redirect psn to the sink psn
00091                                         psn->add(this->p_derived());           // add this to the source subnetwork
00092                                         source->depthFirstExecute(Nmfp,Lmfp,this->psn); // Transmits 'formSubNetwork' on the source side
00093                                 }
00094                         }
00095 
00097                         source->topologyChangeActions();
00098                         sink  ->topologyChangeActions();
00099                         
00100                 }
