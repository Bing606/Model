00030         class ObservingMap{

00032         public:

00034                 typedef typename MapType::iterator staticMapIteratorType;

00036         private:
00037                 static MapType staticMap;               // the static map containing pointers to T objects
00038                 friend class ObservedByMap<KeyType,T,Compare>;  // give ObservedByMap<T> access to staticMap
00039                 
00040         public:
00042                 // Naddresses
00043                 size_t  size() const {
00045                         return this->staticMap.size();
00046                 }       
00047                 
00049                 // address
00050                 T* observed(const KeyType& key) const {
00051                         // element type of std::map are of type pair<const Key,T>
00052                         // return type of find is an iterator iter to a pair<const Key,T>
00053                         // Dereferencing this iterator accesses the element's value, which is of type pair<const Key,T>
00054                         return this->staticMap.find(key)->second;
00055                 }
00056                 
00057                 staticMapIteratorType observedBegin() const {
00058                         return this->staticMap.begin();
00059                 }
00060                 
00061                 
00062                 staticMapIteratorType observedEnd() const {
00063                         return this->staticMap.end();
00064                 }
